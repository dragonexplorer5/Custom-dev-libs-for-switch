üß© Putting It All Together
Your main .nro app will link against these libraries and orchestrate them like this:

cpp
int main() {
    std::vector<uint8_t> save_data;
    std::string save_path = "sdmc:/JKSV/PokemonSword/User1/main";

    if (!load_save(save_path, save_data)) {
        log("Failed to load save.");
        return -1;
    }

    GameID game = detect_game(save_data);
    if (game == GameID::UNKNOWN) {
        log("Unsupported game.");
        return -1;
    }

    Pokemon mon;
    parse_pokemon(save_data, 0x2000, mon); // Example offset
    set_level(mon, 100);
    set_species(mon, 25); // Pikachu

    write_pokemon(save_data, 0x2000, mon);
    backup_save(save_data, game_name(game), "User1");
    export_save(save_data, game_name(game), "User1");

    return 0;
}
